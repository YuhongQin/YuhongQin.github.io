import tkinter as tk
from tkinter import *
from tkinter import Tk, Label
from PIL import ImageTk, Image



class Mancala:

    
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Mancala Game")
        self.buttons = []
        self.buttons0 = []
        
        # self.frame = tk.Frame(self.root,width = 4000,height = 4000)
        self.player = 0
        self.board = [4, 4, 4, 4, 4, 4, 0, 4, 4, 4, 4, 4, 4, 0]
        self.continue_game = True
        self.label1 = tk.Label(self.root, text="You Win!")
        self.label2 = tk.Label(self.root, text="You Lose!")
        self.gamestate = 1
        
        self.label3 = tk.Label(self.root, text="MANCALA",bg="lightblue")
        self.label3.grid(row=0,column=0,padx=20,pady=20)
        self.label4 = tk.Label(self.root, text="",bg="lightblue")
        self.label4.grid(row=5,column=0,padx=20,pady=20)

        self.labelfirst = Label(self.root, text="Your turn",bg="pink",fg='white')
        self.labelsecond = Label(self.root, text="Computer's turn",bg="pink",fg='white')
        self.labelfirst.grid(row=0,column=8,padx=20,pady=20)
        self.root.after(1000,lambda:self.labelfirst.grid_forget())

        
        
        for i in range(15):
            
            if i < 14:
                
                button = tk.Button(self.root, text=str(self.board[i]),relief="ridge")
                # button.bind("<Enter>", lambda event,index=i: self.left_click(event,index))
                button.configure(command=lambda index=i: self.move(index))
                
                if i <6:
                    hex_color = f"#{213:02x}{228:02x}{233:02x}"    
                    button.grid(row=1, column=i+1,padx=20,pady=20,rowspan=2,ipadx=20)
                    button.configure(bg=hex_color)

                elif i == 6:
                    hex_color = f"#{209:02x}{231:02x}{255:02x}"
                    button.grid(row=2,column=7,padx=20,pady=20,rowspan=3,ipady=40,ipadx=20)
                    button.configure(bg=hex_color)

                elif i < 13:
                    hex_color = f"#{255:02x}{240:02x}{232:02x}"
                    button.grid(row=3,column=13 - i,padx=20,pady=20,rowspan=2,ipadx=20)
                    button.configure(bg=hex_color)
                    
                elif i == 13:
                    hex_color = f"#{252:02x}{206:02x}{197:02x}"
                    button.grid(row=2,column=0,padx=20,pady=20,rowspan=3,ipady=40,ipadx=20)
                    button.configure(bg=hex_color)
            else:
                button = tk.Button(self.root,text="restart game",command=lambda:self.restart())
                button.grid(row=7,column=8,padx=20,pady=20,ipadx=20,ipady=20,rowspan=3)
            self.buttons.append(button)
            
            
            
    def left_click(self,event,index):
        CreateToolTip(self.buttons[index],f"next is{(index + self.board[index])%14}")
        

    def move(self, index):
        
        
        if self.continue_game == True:
            if self.player == 0:
                
                
                self.moveforOne(index)
                
            else:
                
                self.moveforTwo(index)

        else:return
    def moveforOne(self,index):
        
        if self.board[index] == 0 or index >=6:
            return
            
        else:
            
            chess = self.board[index]
            self.board[index] = 0
            index += 1
            while chess > 0 :
                index = index % 13
                self.board[index] += 1
                chess -= 1
                index += 1
            index -=1
            if index != 6 and index != 13:
                self.gamestate = 1
                self.eat(index)
            elif index == 6:
                self.gamestate = 2
                self.once_again()
                self.update()
            else:
                self.gamestate = 1
                self.player = (self.player + 1) % 2
                self.update()    
            if(self.gamestate == 1):
                self.labelsecond.grid(row=5,column=8,padx=20,pady=20)
                self.root.after(1000,lambda:self.labelsecond.grid_forget())
    def moveforTwo(self,index):
        
        if self.board[index] == 0 or index <= 6 or index == 13:
            return
            
        else:
            
            chess = self.board[index]
            self.board[index] = 0
            index += 1
            while chess > 0 :
                index = index % 14
                if index == 6:
                    index = 7
                    self.board[index] += 1
                else:
                    self.board[index] += 1
                chess -= 1
                index += 1 
            index -=1
            
            if index != 6 and index != 13:
                self.gamestate = 1
                self.eat(index)

            elif index == 13: 
                self.gamestate = 2
                self.once_again()
                self.update()   
            else:
                self.gamestate = 1
                self.player = (self.player + 1) % 2
                self.update()
            if(self.gamestate == 1):
                self.labelfirst.grid(row=0,column=8,padx=20,pady=20)
                self.root.after(1000,lambda:self.labelfirst.grid_forget())

    def eat(self,index):
        if self.board[index] == 1:
            if self.player == 0:
                opposite = self.board[12 - index % 12]
                self.board[12 - index % 12] = 0
                self.board[6] += opposite + 1
            if self.player == 1:
                opposite = self.board[12 - index]
                self.board[12 - index] = 0
                self.board[13] += opposite + 1
            self.board[index] = 0
        self.player = (self.player + 1) % 2
        self.update()
            

    def check_win(self):
        if self.board[6] >= 25:
            return 0
        elif self.board[13] >= 25:
            return 1
        else:
            return 2
    def update(self):
        for i in range(14):
            self.buttons[i].configure(text=str(self.board[i]))
        if(self.check_win() == 0):
                label = self.label1
                label.grid(row = 7, column = 1)
                self.continue_game = False
        elif(self.check_win() == 1):
                label = self.label2
                label.grid(row = 7, column = 1)
                self.continue_game = False

        # canvas = tk.Canvas(self.root, width=40,height=40,bd=0, highlightthickness=0)

    def once_again(self):
        label = Label(self.root,text="once again!",bg="green",fg='white')
        label.grid(row=2,column=8)
        self.root.after(1000,lambda:label.grid_forget())
        
    def restart(self):
        self.board = [4, 4, 4, 4, 4, 4, 0, 4, 4, 4, 4, 4, 4, 0]
        self.player = 0
        self.update()
        self.continue_game = True
        self.label1.destroy()
        self.label2.destroy()
    def run(self):
        return
    def game(self):
        self.continue_game = True
        self.create_area()
        
        
        self.root.mainloop() 



class ToolTip(object):

    def __init__(self, widget):
        self.widget = widget
        self.tipwindow = None
        self.id = None
        self.x = self.y = 0
    #å½“å…‰æ ‡ç§»åŠ¨æŒ‡å®šæ§ä»¶æ˜¯æ˜¾ç¤ºæ¶ˆæ¯
    def showtip(self, text):
        "Display text in tooltip window"
        self.text = text
        if self.tipwindow or not self.text:
            return
        x, y, cx, cy = self.widget.bbox("insert")
        x = x + self.widget.winfo_rootx()+30
        y = y + cy + self.widget.winfo_rooty()+30
        self.tipwindow = tw = Toplevel(self.widget)
        tw.wm_overrideredirect(1)
        tw.wm_geometry("+%d+%d" % (x, y))
        label = Label(tw, text=self.text,justify=LEFT,
                      background="white", relief=SOLID, borderwidth=1,
                      font=("ä»¿å®‹", "10"))
        label.pack(side=BOTTOM)
    #å½“å…‰æ ‡ç§»å¼€æ—¶æç¤ºæ¶ˆæ¯éšè—
    def hidetip(self):
        tw = self.tipwindow
        self.tipwindow = None
        if tw:
            tw.destroy()

def CreateToolTip(widget, text):
    toolTip = ToolTip(widget)
    def enter(event):
        toolTip.showtip(text)
    def leave(event):
        toolTip.hidetip()
    widget.bind('<Enter>', enter)
    widget.bind('<Leave>', leave)
